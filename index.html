basic_usage.py
from abstractllm import create_llm

# Works with any provider
providers = ["openai", "anthropic", "ollama"]

for provider in providers:
    llm = create_llm(provider)
    response = llm.generate("What is Python?")
    print(f"{provider}: {response.content[:50]}...")

tool_calling.py
from abstractllm import create_llm, tool

@tool
def get_weather(city: str) -> str:
    """Get current weather for a city."""
    return f"Weather in {city}: 72¬∞F, Sunny"

llm = create_llm("openai", model="gpt-4o-mini")
response = llm.generate(
    "What's the weather in Paris?",
    tools=[get_weather]
)
print(response.content)
# Output: The weather in Paris is 72¬∞F and sunny!
structured_output.py
from pydantic import BaseModel
from abstractllm import create_llm

class Person(BaseModel):
    name: str
    age: int
    occupation: str

llm = create_llm("openai", model="gpt-4o-mini")
person = llm.generate(
    "Extract: John Doe is 25 and works as a developer",
    response_model=Person
)

print(f"{person.name}, {person.age}, {person.occupation}")
# Output: John Doe, 25, developer
streaming.py
from abstractllm import create_llm

llm = create_llm("openai", model="gpt-4o-mini")

print("AI: ", end="", flush=True)
for chunk in llm.generate("Write a haiku about coding", stream=True):
    print(chunk.content, end="", flush=True)
print()
